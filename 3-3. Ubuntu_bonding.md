4.&nbsp;네트워크 인터페이스 설정
------------------------------

#### bond0 네트워크 인터페이스 설정 
```
root@server-b:~# cd /etc/netplan/
root@server-b:/etc/netplan# vi 00-installer-config.yaml
network:
  ethernets:
    ens32:
      dhcp4: false
    ens35:
      dhcp4: false
  version: 2

  bonds:
    bond0:
      dhcp4: true
      interfaces: [ens32, ens35]
      parameters:
        mode: active-backup
        mii-monitor-interval: 1

root@server-b:/etc/netplan# netplan apply
```
